#investment strategy Performance
df['Signal'] = 0
df.loc[df['SMA_50'] > df['SMA_200'], 'Signal'] = 1  
df.loc[df['SMA_50'] < df['SMA_200'], 'Signal'] = -1  
df['Position'] = df['Signal'].shift(1)
df['Strategy_Returns'] = df['Position'] * df['Daily_Return']
plt.figure(figsize=(12, 5))
plt.plot((1 + df['Daily_Return']).cumprod(), label="Buy & Hold", color="orange")
plt.plot((1 + df['Strategy_Returns']).cumprod(), label="SMA Strategy", color="blue")
plt.xlabel("Date")
plt.ylabel("Cumulative Returns")
plt.title("Investment Strategy Performance")
plt.legend()
plt.show()
